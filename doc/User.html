def next_image(current_image=nil)
  next_image = nil
  images.each do |i|
    described_by_me = nil
    described_by_me = i.descriptions.detect{ |d| d.user_id == id}

    if current_image
      if described_by_me.nil?  and current_image.id != i.id
        next_image = i
        break
      end
    elsif described_by_me.nil?
      next_image = i
      break
    end

  end
  next_image
end
